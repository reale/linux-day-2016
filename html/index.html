function createDirectory() {
        local directory=${1}
        if [[ ­z “${directory}” ]]; then
                __msg err “argument 1 (directory) missing”
                return 2
        fi
        __msg debug “directory: ${directory}”
        if ! mkdir ­p “${directory}” >>”${_L}” 2>&1; then
                __msg err “failed to create directory '${directory}'”
                return 2
        fi
        return 0
}
